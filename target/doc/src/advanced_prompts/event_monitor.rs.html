39#[derive(Debug)]
40pub enum SystemEvent {
41    PowerSwitchedToAC,
42    PowerSwitchedToBattery,
43    BatteryLevelReport(u8),
44    UsbDeviceConnected,
45    UsbDeviceDisconnected,
46    SystemStartup,
47    SystemShutdown,
48    BatteryInserted,
49    BatteryRemoved,
50    NetworkConnected { ssid: String },
51    NetworkDisconnected,
52}

54pub fn start_monitoring(sender: mpsc::Sender<SystemEvent>) {
55    // 執行緒 1: Win32 消息循環
56    let win32_sender = sender.clone();
57    thread::spawn(move || {
58        if let Err(e) = run_message_loop(win32_sender) {
59            eprintln!("Win32 事件監控執行緒出錯: {:?}", e);
60        }
61    });
62
63    // 執行緒 2: WinRT 電池監控
64    let winrt_battery_sender = sender.clone(); // <-- 這裡也需要 clone
65    thread::spawn(move || {
66        let rt = Runtime::new().expect("建立 Tokio 執行時失敗");
67        rt.block_on(async {
68            if let Err(e) = setup_battery_monitor(winrt_battery_sender).await {
69                eprintln!("設定 WinRT 電池監控失敗: {:?}", e);
70            }
71        });
72    });
73    // 執行緒 3:  WinRT 網路監控
74    let winrt_network_sender = sender; // <-- 最後一個執行緒可以直接移動 sender 的所有權
75    thread::spawn(move || {
76        let rt = Runtime::new().expect("建立 Tokio 執行時失敗");
77        rt.block_on(async {
78            if let Err(e) = setup_network_monitor(winrt_network_sender).await {
79                eprintln!("設定 WinRT 網路監控失敗: {:?}", e);
80            }
81        });
82    });
83}

85// ---  WinRT Network Monitoring (已修正) ---
86async fn setup_network_monitor(sender: mpsc::Sender<SystemEvent>) -> windows::core::Result<()> {
87    let last_connection_state = Arc::new(Mutex::new(None::<String>));
88
89    // FIX 3: 修正輔助函數以正確處理 `GetInternetConnectionProfile` 的 `Result`
90    let get_current_ssid = || -> windows::core::Result<Option<String>> {
91        match NetworkInformation::GetInternetConnectionProfile() {
92            Ok(profile) => Ok(Some(profile.ProfileName()?.to_string())),
93            Err(_) => Ok(None), // 如果 API 傳回錯誤，我們視為沒有連接
94        }
95    };
96
97    // 初始化狀態
98    {
99        let initial_ssid = get_current_ssid()?;
100        *last_connection_state.lock().unwrap() = initial_ssid;
101        println!("[DEBUG] WinRT Monitor: Initial network state: {:?}", last_connection_state.lock().unwrap());
102    }
103
104    // FIX 4: 使用正確的 NetworkStatusChangedEventHandler
105    let handler = NetworkStatusChangedEventHandler::new({
106        let sender_clone = sender.clone();
107        let state_clone = last_connection_state.clone();
108
109        // 這個閉包的簽名現在是 `move |_|`，因為事件本身不傳遞有用的參數
110        move |_| {
111            println!("[DEBUG] WinRT: NetworkStatusChanged event fired!");
112            let current_ssid_opt = get_current_ssid()?;
113            let mut last_ssid_guard = state_clone.lock().unwrap();
114
115            if *last_ssid_guard != current_ssid_opt {
116                println!("[DEBUG]   - Network state changed from {:?} to {:?}", *last_ssid_guard, current_ssid_opt);
117                match (last_ssid_guard.as_ref(), current_ssid_opt.as_ref()) {
118                    (None, Some(new_ssid)) => {
119                        sender_clone.send(SystemEvent::NetworkConnected { ssid: new_ssid.clone() }).unwrap();
120                    },
121                    (Some(_), None) => {
122                        sender_clone.send(SystemEvent::NetworkDisconnected).unwrap();
123                    },
124                    (Some(old_ssid), Some(new_ssid)) if old_ssid != new_ssid => {
125                        sender_clone.send(SystemEvent::NetworkDisconnected).unwrap();
126                        sender_clone.send(SystemEvent::NetworkConnected { ssid: new_ssid.clone() }).unwrap();
127                    },
128                    _ => {}
129                }
130                *last_ssid_guard = current_ssid_opt;
131            }
132            Ok(())
133        }
134    });
135
136    let token = NetworkInformation::NetworkStatusChanged(&handler)?;
137    println!("[INFO] 成功訂閱 WinRT NetworkStatusChanged 事件。");
138    
139    std::future::pending::<()>().await;
140
141    NetworkInformation::RemoveNetworkStatusChanged(token)?;
142    Ok(())
143}

145// --- WinRT Battery Monitoring ---
146async fn setup_battery_monitor(sender: mpsc::Sender<SystemEvent>) -> windows::core::Result<()> {
147    let aggregate_battery = Battery::AggregateBattery()?;
148    let last_present_state = Arc::new(Mutex::new(None::<bool>));
149
150    // 初始化狀態
151    {
152        let report = aggregate_battery.GetReport()?;
153        let is_present = match report.FullChargeCapacityInMilliwattHours() {
154            Ok(capacity_opt) => capacity_opt.GetInt32().map_or(false, |cap| cap > 0),
155            Err(_) => false,
156        };
157        *last_present_state.lock().unwrap() = Some(is_present);
158        println!("[DEBUG] WinRT Monitor: Initial battery present state: {}", is_present);
159    }
160    
161    let handler = TypedEventHandler::<Battery, IInspectable>::new({
162        let sender_clone = sender.clone();
163        let state_clone = last_present_state.clone();
164        
165        move |battery_sender, _| {
166            println!("[DEBUG] WinRT: Battery.ReportUpdated event fired!");
167            
168            let battery = match battery_sender.as_ref() {
169                Some(b) => b,
170                None => return Ok(()),
171            };
172
173            let report_result = battery.GetReport();
174            let is_present = match report_result {
175                Ok(report) => {
176                    match report.FullChargeCapacityInMilliwattHours() {
177                        Ok(capacity_opt) => capacity_opt.GetInt32().map_or(false, |cap| cap > 0),
178                        Err(_) => false,
179                    }
180                },
181                Err(_) => false,
182            };
183            
184            let mut was_present_guard = state_clone.lock().unwrap();
185            let was_present_opt = *was_present_guard;
186
187            if was_present_opt.is_none() || was_present_opt.unwrap() == is_present {
188                // 狀態未變或首次運行
189            } else {
190                let was_present = was_present_opt.unwrap();
191                if was_present && !is_present {
192                    println!("[DEBUG]   >>>> CHANGE DETECTED: Battery Removed! Sending event...");
193                    let _ = sender_clone.send(SystemEvent::BatteryRemoved);
194                } else if !was_present && is_present {
195                    println!("[DEBUG]   >>>> CHANGE DETECTED: Battery Inserted! Sending event...");
196                    let _ = sender_clone.send(SystemEvent::BatteryInserted);
197                }
198            }
199            *was_present_guard = Some(is_present);
200            Ok(())
201        }
202    });
203
204    aggregate_battery.ReportUpdated(&handler)?;
205    println!("[INFO] 成功訂閱 WinRT Battery.ReportUpdated 事件。");
206
207    std::future::pending::<()>().await;
208
209    Ok(())
210}

212// --- Win32 Message Loop ---
213fn run_message_loop(sender: mpsc::Sender<SystemEvent>) -> windows::core::Result<()> {
214    let class_name = w!("AdvancedPromptsEventMonitorClass");
215    let instance = unsafe { GetModuleHandleW(None)? };
216
217    let wc = WNDCLASSW {
218        lpfnWndProc: Some(wndproc),
219        hInstance: instance.into(),
220        lpszClassName: class_name,
221        ..Default::default()
222    };
223    unsafe { if RegisterClassW(&wc) == 0 { return Err(windows::core::Error::from_win32()); } }
224
225    let sender_box = Box::new(sender);
226    let sender_ptr = Box::into_raw(sender_box) as *mut c_void;
227
228    let hwnd = unsafe {
229        CreateWindowExW(
230            Default::default(), class_name, w!("AP_Event_Window"), Default::default(),
231            CW_USEDEFAULT, CW_USEDEFAULT, CW_USEDEFAULT, CW_USEDEFAULT,
232            Some(HWND_MESSAGE), None, Some(instance.into()), Some(sender_ptr),
233        )?
234    };
235
236    let mut msg = MSG::default();
237    while unsafe { GetMessageW(&mut msg, None, 0, 0) }.as_bool() {
238        unsafe { TranslateMessage(&msg); DispatchMessageW(&msg); }
239    }
240    unsafe { UnregisterClassW(class_name, Some(instance.into())); }
241    Ok(())
242}

244extern "system" fn wndproc(window: HWND, message: u32, wparam: WPARAM, lparam: LPARAM) -> LRESULT {
245    match message {
246        WM_CREATE => {
247            let create_struct = unsafe { &*(lparam.0 as *const windows::Win32::UI::WindowsAndMessaging::CREATESTRUCTW) };
248            let sender_ptr = create_struct.lpCreateParams;
249            unsafe { SetWindowLongPtrW(window, GWLP_USERDATA, sender_ptr as isize); }
250
251            let _power_handle = unsafe { RegisterPowerSettingNotification(window.into(), &GUID_ACDC_POWER_SOURCE, REGISTER_NOTIFICATION_FLAGS(0)) };
252            let mut notification_filter = DEV_BROADCAST_DEVICEINTERFACE_W {
253                dbcc_size: std::mem::size_of::<DEV_BROADCAST_DEVICEINTERFACE_W>() as u32,
254                dbcc_devicetype: DBT_DEVTYP_DEVICEINTERFACE.0,
255                dbcc_classguid: GUID_DEVINTERFACE_USB_DEVICE,
256                ..Default::default()
257            };
258            let _device_handle = unsafe { RegisterDeviceNotificationW(window.into(), &mut notification_filter as *mut _ as *mut c_void, DEVICE_NOTIFY_WINDOW_HANDLE) };
259            
260            if let Some(sender) = get_sender(window) {
261                sender.send(SystemEvent::SystemStartup).unwrap();
262            }
263            LRESULT(0)
264        }
265        WM_POWERBROADCAST => {
266            if wparam.0 as u32 == PBT_POWERSETTINGCHANGE {
267                let sender = match get_sender(window) { Some(s) => s, None => return LRESULT(0), };
268                let setting = unsafe { &*(lparam.0 as *const POWERBROADCAST_SETTING) };
269                if setting.PowerSetting == GUID_ACDC_POWER_SOURCE {
270                    let status = unsafe { *(setting.Data.as_ptr() as *const u32) };
271                    if status == 0 {
272                        sender.send(SystemEvent::PowerSwitchedToAC).unwrap();
273                    } else {
274                        sender.send(SystemEvent::PowerSwitchedToBattery).unwrap();
275                        let mut sps = SYSTEM_POWER_STATUS::default();
276                        if unsafe { GetSystemPowerStatus(&mut sps) }.is_ok() && sps.BatteryLifePercent != 255 {
277                            sender.send(SystemEvent::BatteryLevelReport(sps.BatteryLifePercent)).unwrap();
278                        }
279                    }
280                }
281            }
282            LRESULT(0)
283        }
284        WM_DEVICECHANGE => {
285            let event_type = wparam.0 as u32;
286            if event_type == DBT_DEVICEARRIVAL || event_type == DBT_DEVICEREMOVECOMPLETE {
287                let header = unsafe { &*(lparam.0 as *const DEV_BROADCAST_HDR) };
288                if header.dbch_devicetype == DBT_DEVTYP_DEVICEINTERFACE {
289                     if let Some(sender) = get_sender(window) {
290                         let event = if event_type == DBT_DEVICEARRIVAL { SystemEvent::UsbDeviceConnected } else { SystemEvent::UsbDeviceDisconnected };
291                         sender.send(event).unwrap();
292                     }
293                }
294            }
295            LRESULT(0)
296        }
297        WM_DESTROY => {
298            let sender_ptr = unsafe { SetWindowLongPtrW(window, GWLP_USERDATA, 0) };
299            if sender_ptr != 0 {
300                unsafe { Box::from_raw(sender_ptr as *mut mpsc::Sender<SystemEvent>); }
301            }
302            unsafe { PostQuitMessage(0) };
303            LRESULT(0)
304        }
305        _ => unsafe { DefWindowProcW(window, message, wparam, lparam) },
306    }
307}

309fn get_sender<'a>(window: HWND) -> Option<&'a mpsc::Sender<SystemEvent>> {
310    let ptr = unsafe { GetWindowLongPtrW(window, GWLP_USERDATA) };
311    if ptr == 0 { None } else { unsafe { Some(&*(ptr as *const mpsc::Sender<SystemEvent>)) } }
