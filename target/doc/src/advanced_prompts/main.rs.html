18fn get_windows_username() -> String {
19    let mut buffer = [0u16; 256];
20    let mut size = buffer.len() as u32;
21
22    unsafe {
23        if GetUserNameW(Some(PWSTR(buffer.as_mut_ptr())), &mut size).is_ok() {
24            String::from_utf16_lossy(&buffer[..size as usize])
25        } else {
26            "user".to_string()
27        }
28    }
29}
30
31fn main() {
32    println!("“高级提示”应用启动...");
33
34    let username = get_windows_username();
35    println!("侦测到目前使用者: {}", username);
36
37    let locale = "en";
38    let i18n = I18nManager::new(locale).expect("语言文件加载失败");
39    let mut tts = TtsEngine::new(locale).expect("语音引擎初始化失败");
40
41    let (sender, receiver) = mpsc::channel();
42    
43    start_monitoring(sender);
44
45    println!("事件监视器已启动，等待系统事件...");
46
47    for received_event in receiver {
48        println!("接收到事件: {:?}", received_event);
49        
50        let event_ref = &received_event;
51        let text_to_speak = match event_ref {
52            SystemEvent::SystemStartup => {
53                i18n.get_text_with_param("system_online", "user", &username)
54            },
55            
56            SystemEvent::PowerSwitchedToAC => i18n.get_text("external_power_connected"),
57            SystemEvent::PowerSwitchedToBattery => i18n.get_text("switched_to_battery"),
58            SystemEvent::BatteryLevelReport(level) => {
59                if *level < 20 {
60                    i18n.get_text("battery_level_critical")
61                } else {
62                    i18n.get_text_with_param("battery_level_report", "level", &level.to_string())
63                }
64            },
65            SystemEvent::UsbDeviceConnected => i18n.get_text("usb_device_detected"),
66            SystemEvent::UsbDeviceDisconnected => i18n.get_text("usb_device_disconnected"),
67            SystemEvent::SystemShutdown => i18n.get_text("system_shutdown"),
68            
69            // --- 修改部分开始 ---
70            SystemEvent::BatteryInserted => {
71                // 创建一个可变的电源状态结构体
72                let mut sps = SYSTEM_POWER_STATUS::default();
73                // 调用 API 查询状态
74                if unsafe { GetSystemPowerStatus(&mut sps) }.is_ok() {
75                    // 检查电池电量是否有效 (255 表示无效)
76                    if sps.BatteryLifePercent != 255 {
77                        // 使用带参数的函数来填充提示语
78                        i18n.get_text_with_param("battery_inserted", "level", &sps.BatteryLifePercent.to_string())
79                    } else {
80                        // 如果无法获取电量，就使用一个不带参数的回退版本
81                        // (您可以在 en.json 中添加一个 "battery_inserted_error" 键)
82                        i18n.get_text("battery_inserted_error") 
83                    }
84                } else {
85                    // 如果 API 调用失败，也使用回退版本
86                    i18n.get_text("battery_inserted_error")
87                }
88            },
89            // --- 修改部分结束 ---
90
91            SystemEvent::BatteryRemoved => i18n.get_text("battery_removed"),
92        };
93
94        if let Some(text) = text_to_speak {
95            println!("准备播报: {}", text);
96            if let Err(e) = tts.speak(&text) {
97                eprintln!("语音播报失败: {}", e);
98            }
99        } else {
100            println!("[ERROR] 未能为事件 {:?} 找到对应的提示语文字！请检查 locales/en.json 檔案。", event_ref);
101        }
102    }
103}
